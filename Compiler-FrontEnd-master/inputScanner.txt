 * para o analisador léxico JFlex.
 */

void principal(int x, float y) { // Declaração da função principal
    int contador = 0;
    float resultado = 0.0;
    
    // Laço de repetição
    while (contador < 10) {
        resultado += 1.5; // Operador composto
        contador++;     // Operador de incremento
    }
    
    if (x == 5) {
        resultado = 99; // Atribuição de inteiro (deve ser INT)
    } else {
        resultado = 123.45; // Atribuição de float (deve ser FLOAT)
    }

    return resultado;
}
